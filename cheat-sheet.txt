
sudo rm -f /opt/rootfs_debootstrap/dev/null
sudo mknod -m 666 /opt/rootfs_debootstrap/dev/null c 1 3

sudo mkdir -p /opt/rootfs_debootstrap/{proc,sys,dev,run,tmp}
sudo chmod 1777 /opt/rootfs_debootstrap/tmp

sudo mount -t proc /proc /opt/rootfs_debootstrap/proc
sudo mount --rbind /sys /opt/rootfs_debootstrap/sys
sudo mount --rbind /dev /opt/rootfs_debootstrap/dev
sudo mount --rbind /run /opt/rootfs_debootstrap/run

sudo chroot /opt/rootfs_debootstrap /bin/bash

mkdir -p /etc/apt/apt.conf.d

cat > /etc/apt/apt.conf.d/99insecure <<'EOF'
Acquire::AllowInsecureRepositories "true";
Acquire::AllowDowngradeToInsecureRepositories "true";
Acquire::https::Verify-Peer "false";
Acquire::https::Verify-Host "false";

APT::Get::AllowUnauthenticated "true";
APT::Get::AllowDowngradeToInsecureRepositories "true";

Debug::Acquire::gpgv "false";
EOF

nano /etc/apt/sources.list

deb [trusted=yes] http://deb.debian.org/debian bookworm main contrib non-free
deb [trusted=yes] http://deb.debian.org/debian bookworm-updates main contrib non-free
deb [trusted=yes] http://security.debian.org/debian-security bookworm-security main contrib non-free	

mkdir -p /tmp
chmod 1777 /tmp

echo test > /dev/null

apt update

apt install -y python3.10 python3-pip bash curl ca-certificates

# Remove apt cache (biggest space saver)
apt clean
rm -rf /var/lib/apt/lists/*
rm -rf /var/cache/apt/*

# Remove docs, man pages
rm -rf /usr/share/doc/*
rm -rf /usr/share/man/*

# Remove all locales except minimal
rm -rf /usr/share/locale/*
mkdir -p /usr/share/locale/en_US

# Remove dev junk (optional but big savings)
rm -rf /usr/include/*
rm -rf /usr/src/*

# Remove temp
rm -rf /tmp/*,*

exit
